70 beamformer::get_e(float *e_l, float *e_r, const float *x_l, const float *x_r, size_t ref_size) {
71  std::shared_ptr<beamformer_param_t> data_current = std::atomic_load(&currentParam);
72  if (!data_current->bf_on_off_) {
73  std::memcpy(e_l,x_l,ref_size*sizeof(float));
74  std::memcpy(e_r,x_r,ref_size*sizeof(float));
75  return;
76  }
77  // compute u(n) and v(n)
78  for (size_t i = 0; i < ref_size; i++) {
79  u[i] = (x_l[i] - x_r[i]) * 0.5f;
80  v[i] = (x_l[i] + x_r[i]) * 0.5f;
81  }
82  // compute y(n)
83  this->cirfir(u, y, ref_size);
84  // compute d(n)
85  delay_buffer_->set(v, ref_size); // put a new frame to buffer
86  size_t delay_head_backup = delay_buffer_->head_; // backup the position of current head
87  delay_buffer_->head_ =
88  (delay_buffer_->head_ - delay_len_) & delay_buffer_->mask_; // shift the position of head by delay_len_
89  delay_buffer_->get(d, ref_size); // get d(n)
90  delay_buffer_->head_ = delay_head_backup; // restore the head
91  // compute e(n)
92  for (size_t i = 0; i < ref_size; i++) {
93  e[i] = d[i] - y[i];
94  }
95 
96  std::memcpy(e_l,e,ref_size*sizeof(float));
97  std::memcpy(e_r,e,ref_size*sizeof(float));
98 }

101 int
102 beamformer::update_bf_taps(size_t ref_size) {
103  std::shared_ptr<beamformer_param_t> data_current = std::atomic_load(&currentParam);
104  if (!data_current->bf_on_off_) {
105  return 0;
106  }
107  // adaptation flag
108  int adaptation_on_off = 1;
109 
110  // compute the power estimate for the adaptation mode controller
111  float amc_forgetting_factor = data_current->amc_forgetting_factor_;
112  pe_u_ = amc_forgetting_factor * pe_u_ + (1 - amc_forgetting_factor) * (array_mean_square(u, ref_size));
113  pe_v_ = amc_forgetting_factor * pe_v_ + (1 - amc_forgetting_factor) * (array_mean_square(v, ref_size));
114 
115  // apply the adaptation mode controller
116  if(data_current->bf_amc_on_off_) {
117  float ratio = pe_v_/pe_u_;
118  if(ratio>data_current->amc_thr_) {
119  adaptation_on_off = 0;
120  }
121  }
122 
123  // update the taps of the adaptive filter
124  if(adaptation_on_off) {
125  this->update_taps(u, e, ref_size);
126  // norm-constrained adaptation for beamforming
127  if(data_current->bf_nc_on_off_) {
128  size_t taps_len = this->get_size();
129  this->get_taps(w, taps_len);
130  float omg = array_dot_product(w, w, taps_len);
131  if(omg>data_current->nc_thr_){
132  float scaling_factor = std::sqrt(data_current->nc_thr_/omg);
133  array_multiply_const(w, scaling_factor, taps_len);
134  this->set_taps(w, taps_len);
135  }
136  }
137  }
138  return adaptation_on_off;
139 }
