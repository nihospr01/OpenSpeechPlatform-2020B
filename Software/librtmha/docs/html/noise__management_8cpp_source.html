120  /* Peak detect */
121  if (xabs >= peak[n - 1])

125 
126  /* Valley detect */
127

129  valley[n] = aLPv * valley[n - 1] + (1 - aLPv) * xabs;
130  else
131  valley[n] = valley[n - 1] / rLPv;
132  }
133 
134  array_square(peak, xpow, in_len);
135 
136  /* Noise power estimation */
137 
138  npow[0] = (npow_init == 0) ? xpow[0] : npow_init;
139  array_square(valley, vpow, in_len);
140 
141  if (ntype == 1) {
142  /* Power estimation using limits on change (ref: Arslan et al.) */
143  for (size_t n = 1; n < in_len; n++) {
144  n1 = (xpow[n] < a1 * npow[n - 1]) ? xpow[n] : a1 * npow[n - 1];
145  npow[n] = (n1 > a2 * npow[n - 1]) ? n1 : a2 * npow[n - 1];
146  }
147 
148  } else if (ntype == 2) {
149  /* Noise power estimation using the weighted averaging of Hirsch and Ehrlicher */
150  for (size_t n = 1; n < in_len; n++) {
151  /* Update noise average if new signal sample is close to the noise */
152  if (xpow[n] < bHE * npow[n - 1])
153  npow[n] = aHE * npow[n - 1] + (1 - aHE) * xpow[n];
154  /* Otherwise keep the previous noise estimate */
155  else

157 
158  }
159  } else if (ntype == 3) {
160  /* Noise power estimation using MCRA of Cohen and Berdugo */
161  // prob = 0.5; // Prob of first sample being speech
162  for (size_t n = 1; n < in_len; n++) {
163  // P[speech] for this sample
164  if (xpow[n] > delta * vpow[n])
165  p = 1;
166  else

171
