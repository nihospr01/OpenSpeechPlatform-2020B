60  // Compute the compression gain in dB using the peak detector dB values and write compressed signal in out
61  for (size_t i = 0; i < in_len; i++) {
62  if (pdb[i] < knee_low_) {
63  g = glow_; // Linear gain below lower kneepoint
64  } else {
65  g = glow_ + data_current->slope_ * (pdb[i] - knee_low_); // Compression in between kneepoints
66  }
67 
68  if ((pdb[i] + g) >= mpo_limit_) {
69  g = mpo_limit_ - pdb[i]; // Above MPO limit
70 
71  }
72
