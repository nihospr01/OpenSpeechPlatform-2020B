150 void tenband_filterbank::process(float *in, float **out, size_t frame_size) {
151  bool aligned = aligned_;
152  float *in_32 = in;
153  //Downsampling 32->16->8->4
154  downsample[0]->process(in_32, in_16, frame_size);
155  downsample[1]->process(in_16, in_8, frame_size >> 1);
156  downsample[2]->process(in_8, in_4, frame_size >> 2);
157  //Filtering the 4kHz domain & Upsampling from 4kHz->8kHz->16kHz->32kHz & Delay for alignment
158  for (int i = 0; i < 4; i++) {
159  filters[i]->process(in_4, out_4[i], frame_size >> 3);
160  stage1_upsample[i]->process(out_4[i], out_8[i], frame_size >> 3);
161  stage2_upsample[i]->process(out_8[i], out_16[i], frame_size >> 2);
162  stage3_upsample[i]->process(out_16[i], out[i], frame_size >> 1);
163  if(aligned) {
164  delay_blocks[i]->set(out[i], frame_size);
165  delay_blocks[i]->delay_block(out[i], frame_size, harris_align[i]);
166  }
167  }
168  //Filtering the 8kHz domain & Upsampling from 8kHz->16kHz->32kHz & Delay for alignment
169  for (int i = 4; i < 6; i++) {
170  filters[i]->process(in_8, out_8[i], frame_size >> 2);
171  stage1_upsample[i]->process(out_8[i], out_16[i], frame_size >> 2);
172  stage2_upsample[i]->process(out_16[i], out[i], frame_size >> 1);
173  if(aligned) {
174  delay_blocks[i]->set(out[i], frame_size);
175  delay_blocks[i]->delay_block(out[i], frame_size, harris_align[i]);
176  }
177  }
178 
179  //Filtering the 16kHz domain & Upsampling from 16kHz->32kHz & Delay for alignment
180  for (int i = 6; i < 8; i++) {
181  filters[i]->process(in_16, out_16[i], frame_size >> 1);
182  stage1_upsample[i]->process(out_16[i], out[i], frame_size >> 1);
183  if(aligned) {
184  delay_blocks[i]->set(out[i], frame_size);
185  delay_blocks[i]->delay_block(out[i], frame_size, harris_align[i]);
186  }
187  }
188  //Filtering the 32kHz domain & Delay for alignment
189  for (int i = 8; i < 10; i++) {
190  filters[i]->process(in_32, out[i], frame_size);
191  if(aligned) {
192  delay_blocks[i]->set(out[i], frame_size);
193  delay_blocks[i]->delay_block(out[i], frame_size, harris_align[i]);
194  }
195  }
196 }
