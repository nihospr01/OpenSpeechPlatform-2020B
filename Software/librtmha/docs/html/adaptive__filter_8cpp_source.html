70 adaptive_filter::update_taps(float *u_ref, float *e_ref, size_t ref_size) {
71  // Are you adapting the filter taps?
72  std::shared_ptr<adaptive_filter_param_t> data_current = std::atomic_load(&currentParam);
73  if (data_current->adaptation_type_ <= 0) {
74  return 0;
75  }
76  // check the size of reference signal
77  if (ref_size > max_frame_size_) {
78  return -1;
79  }
80  // compute the power estimate of the input to coefficient adaptation
81  data_current->power_estimate_ = data_current->rho_ * data_current->power_estimate_ +
82  (1 - data_current->rho_) * (array_mean_square(u_ref, ref_size) + array_mean_square(e_ref, ref_size));
83  // normalize the step size
84  float mu = data_current->mu_ / (adaptive_filter_tap_len_ * data_current->power_estimate_ + data_current->delta_);
85  // compute the gradient, gradient = [u_ref(n)]*e_ref(n)
86  u_ref_buf_->set(u_ref, ref_size); // put new u_ref frame to the buffer
87  size_t u_ref_buf_head_backup = u_ref_buf_->head_; // backup the original head position
88  size_t mask = u_ref_buf_->mask_;
89  for (size_t i = 0; i < adaptive_filter_tap_len_; i++) {
90  u_ref_buf_->head_ = (u_ref_buf_head_backup - i) & mask; // shift the head position to assess next frame
91  u_ref_buf_->get(u_ref_frame_buf, ref_size); // get the frame data
92  gradient_[i] = array_dot_product(u_ref_frame_buf, e_ref, ref_size); // compute dot product (cross-correlation)
93  }
94  u_ref_buf_->head_ = u_ref_buf_head_backup;
95 
96  // compute the step-size control matrix, S(n), namely step_size_weights
97  int get_check = this->get_taps(filter_taps_,
98  adaptive_filter_tap_len_); // get the current filter taps to compute S(n)
99  // case 1, Modified LMS, S(n) = I (identity)
100  // case 2, IPNLMS-l_0, S(n) = (1-alpha)/2+(1+alpha)|h_i(n)|/[(2/M)one_norm(h(n))]
101  // case 3, SLMS, S(n) = |h_i(n)|^(2-p)/|(1/M)*sum_over_j(|h_j(n)|^(2-p))|
102  switch (data_current->adaptation_type_) {
103  case 1:
104  // no need to compute S(n) for Modified LMS
105  break;
106  case 2:
107  // compute S(n) for IPNLMS-l_0
108  get_step_size_weights_IPNLMS(filter_taps_, step_size_weights_, data_current->alpha_, data_current->beta_, data_current->delta_,
109  adaptive_filter_tap_len_); // get S(n)
110  array_element_multiply_array(gradient_, step_size_weights_, adaptive_filter_tap_len_); // S(n)*u(n)e(n)
111  break;
112  case 3:
113  // compute S(n) for SLMS
114  get_step_size_weights_SLMS(filter_taps_, step_size_weights_, data_current->p_, data_current->c_, adaptive_filter_tap_len_); // get S(n)
115  array_element_multiply_array(gradient_, step_size_weights_, adaptive_filter_tap_len_); // S(n)*u(n)e(n)
116  break;
117  default:
118  break;
119  }
120  array_multiply_const(gradient_, mu, adaptive_filter_tap_len_); // mu*S(n)u(n)e(n)
121  array_add_array(filter_taps_, gradient_, adaptive_filter_tap_len_); // w(n) + mu*S(n)u(n)e(n)
122  int set_check = this->set_taps(filter_taps_, adaptive_filter_tap_len_); // update the filter taps
123  return get_check | set_check;
124 }

170 adaptive_filter::get_step_size_weights_IPNLMS(float *taps, float *step_size_weights, float alpha, float beta,
171  float delta, size_t tap_len) {
172  float tmp_sum = 0;
173  for (size_t i = 0; i < tap_len; i++) {
174  step_size_weights[i] = 1.0f - std::exp(-beta * fabsf(taps[i]));
175  tmp_sum = tmp_sum + step_size_weights[i];
176  }
177  tmp_sum = tmp_sum * 2.0f / (float) tap_len + delta;
178  tmp_sum = (1.0f + alpha) / tmp_sum;
179  array_multiply_const(step_size_weights, tmp_sum, tap_len);
180  array_add_const(step_size_weights, (1.0f - alpha) / 2.0f, tap_len);
181 }

184 adaptive_filter::get_step_size_weights_SLMS(float *taps, float *step_size_weights, float p, float c, size_t tap_len) {
185  float tmp = 0;
186  for (size_t i = 0; i < tap_len; i++) {
187  step_size_weights[i] = powf(fabsf(taps[i]) + c, 2 - p);
188  }
189  tmp = 1.0f / array_mean(step_size_weights, tap_len);
190  array_multiply_const(step_size_weights, tmp, tap_len);
191 }
