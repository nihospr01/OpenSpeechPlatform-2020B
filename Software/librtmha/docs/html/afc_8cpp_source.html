63 afc::get_y_hat(float *y_hat, float *e, float *s, size_t ref_size) {
64  std::shared_ptr<afc_param_t> data_current = std::atomic_load(&currentParam);
65  if (!data_current->afc_on_off_) {
66  for (size_t i = 0; i < ref_size; i++) {
67  y_hat[i] = 0.0f;
68  }
69  return 0;
70  }
71  if (ref_size > this->get_max_frame_size()) {
72  return -1;
73  }
74  // filtering
75  prefilter_u_->cirfir(u_, u_ref, ref_size); // compute u_ref, we need this filtering happen one step forward
76  // delay_block line
77  delay_buffer_->set(s, ref_size); // put a new frame to buffer
78  size_t delay_head_backup = delay_buffer_->head_; // backup the position of current head
79  delay_buffer_->head_ = delay_buffer_->head_ - data_current->delay_len_ & delay_buffer_->mask_; // shift the position of head by delay_len_
80  delay_buffer_->get(u_, ref_size); // to update the current u(n)
81  delay_buffer_->head_ = delay_head_backup; // restore the head
82  // filtering
83  prefilter_e_->cirfir(e, e_ref, ref_size);
84  bandlimited_filter_->cirfir(s, u_, ref_size);
85  int check = this->update_taps(u_ref, e_ref, ref_size);
86  this->cirfir(u_, y_hat, ref_size);
87  return check;
88 }
