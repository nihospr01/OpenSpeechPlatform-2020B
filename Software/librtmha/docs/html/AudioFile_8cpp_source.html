257 bool AudioFile<T>::decodeWaveFile (std::vector<uint8_t>& fileData)
258 {
259  // -----------------------------------------------------------
260  // HEADER CHUNK
261  std::string headerChunkID (fileData.begin(), fileData.begin() + 4);
262  //int32_t fileSizeInBytes = fourBytesToInt (fileData, 4) + 8;
263  std::string format (fileData.begin() + 8, fileData.begin() + 12);
264 
265  // -----------------------------------------------------------
266  // try and find the start points of key chunks
267  int indexOfDataChunk = getIndexOfString (fileData, "data");
268  int indexOfFormatChunk = getIndexOfString (fileData, "fmt");
269 
270  // if we can't find the data or format chunks, or the IDs/formats don't seem to be as expected
271  // then it is unlikely we'll able to read this file, so abort
272  if (indexOfDataChunk == -1 || indexOfFormatChunk == -1 || headerChunkID != "RIFF" || format != "WAVE")
273  {
274  std::cout << "ERROR: this doesn't seem to be a valid .WAV file" << std::endl;
275  return false;
276  }
277 
278  // -----------------------------------------------------------
279  // FORMAT CHUNK
280  int f = indexOfFormatChunk;
281  std::string formatChunkID (fileData.begin() + f, fileData.begin() + f + 4);
282  //int32_t formatChunkSize = fourBytesToInt (fileData, f + 4);
283  int16_t audioFormat = twoBytesToInt (fileData, f + 8);
284  int16_t numChannels = twoBytesToInt (fileData, f + 10);
285  sampleRate = (uint32_t) fourBytesToInt (fileData, f + 12);
286  int32_t numBytesPerSecond = fourBytesToInt (fileData, f + 16);
287  int16_t numBytesPerBlock = twoBytesToInt (fileData, f + 20);
288  bitDepth = (int) twoBytesToInt (fileData, f + 22);
289 
290  int numBytesPerSample = bitDepth / 8;
291 
292  // check that the audio format is PCM
293  if (audioFormat != 1)
294  {
295  std::cout << "ERROR: this is a compressed .WAV file and this library does not support decoding them at present" << std::endl;
296  return false;
297  }
298 
299  // check the number of channels is mono or stereo
300  if (numChannels < 1 ||numChannels > 2)
301  {
302  std::cout << "ERROR: this WAV file seems to be neither mono nor stereo (perhaps multi-track, or corrupted?)" << std::endl;
303  return false;
304  }
305 
306  // check header data is consistent
307  if ((numBytesPerSecond != (numChannels * sampleRate * bitDepth) / 8) || (numBytesPerBlock != (numChannels * numBytesPerSample)))
308  {
309  std::cout << "ERROR: the header data in this WAV file seems to be inconsistent" << std::endl;
310  return false;
311  }
312 
313  // check bit depth is either 8, 16 or 24 bit
314  if (bitDepth != 8 && bitDepth != 16 && bitDepth != 24)
315  {
316  std::cout << "ERROR: this file has a bit depth that is not 8, 16 or 24 bits" << std::endl;
317  return false;
318  }
319 
320  // -----------------------------------------------------------
321  // DATA CHUNK
322  int d = indexOfDataChunk;
323  std::string dataChunkID (fileData.begin() + d, fileData.begin() + d + 4);
324  int32_t dataChunkSize = fourBytesToInt (fileData, d + 4);
325 
326  int numSamples = dataChunkSize / (numChannels * bitDepth / 8);
327  int samplesStartIndex = indexOfDataChunk + 8;
328 
329  clearAudioBuffer();
330  samples.resize (numChannels);
331 
332  for (int i = 0; i < numSamples; i++)
333  {
334  for (int channel = 0; channel < numChannels; channel++)
335  {
336  int sampleIndex = samplesStartIndex + (numBytesPerBlock * i) + channel * numBytesPerSample;
337 
338  if (bitDepth == 8)
339  {
340  T sample = singleByteToSample (fileData[sampleIndex]);
341  samples[channel].push_back (sample);
342  }
343  else if (bitDepth == 16)
344  {
345  int16_t sampleAsInt = twoBytesToInt (fileData, sampleIndex);
346  T sample = sixteenBitIntToSample (sampleAsInt);
347  samples[channel].push_back (sample);
348  }
349  else if (bitDepth == 24)
350  {
351  int32_t sampleAsInt = 0;
352  sampleAsInt = (fileData[sampleIndex + 2] << 16) | (fileData[sampleIndex + 1] << 8) | fileData[sampleIndex];
353 
354  if (sampleAsInt & 0x800000) // if the 24th bit is set, this is a negative number in 24-bit world
355  sampleAsInt = sampleAsInt | ~0xFFFFFF; // so make sure sign is extended to the 32 bit float
356 
357  T sample = (T)sampleAsInt / (T)8388608.;
358  samples[channel].push_back (sample);
359  }
360  else
361  {
362  assert (false);
363  }
364  }
365  }
366 
367  return true;
368 }

376  std::string headerChunkID (fileData.begin(), fileData.begin() + 4);
377  //int32_t fileSizeInBytes = fourBytesToInt (fileData, 4, Endianness::BigEndian) + 8;
378  std::string format (fileData.begin() + 8, fileData.begin() + 12);
379 
380  // -----------------------------------------------------------
381  // try and find the start points of key chunks
382  int indexOfCommChunk = getIndexOfString (fileData, "COMM");
383  int indexOfSoundDataChunk = getIndexOfString (fileData, "SSND");
384 
385  // if we can't find the data or format chunks, or the IDs/formats don't seem to be as expected
386  // then it is unlikely we'll able to read this file, so abort
387  if (indexOfSoundDataChunk == -1 || indexOfCommChunk == -1 || headerChunkID != "FORM" || format != "AIFF")
388  {
389  std::cout << "ERROR: this doesn't seem to be a valid AIFF file" << std::endl;
390  return false;
391  }
392 
393  // -----------------------------------------------------------
394  // COMM CHUNK
395  int p = indexOfCommChunk;
396  std::string commChunkID (fileData.begin() + p, fileData.begin() + p + 4);
397  //int32_t commChunkSize = fourBytesToInt (fileData, p + 4, Endianness::BigEndian);
398  int16_t numChannels = twoBytesToInt (fileData, p + 8, Endianness::BigEndian);
399  int32_t numSamplesPerChannel = fourBytesToInt (fileData, p + 10, Endianness::BigEndian);
400  bitDepth = (int) twoBytesToInt (fileData, p + 14, Endianness::BigEndian);
401  sampleRate = getAiffSampleRate (fileData, p + 16);
402 
403  // check the sample rate was properly decoded
404  if (sampleRate == -1)
405  {
406  std::cout << "ERROR: this AIFF file has an unsupported sample rate" << std::endl;
407  return false;
408  }
409 
410  // check the number of channels is mono or stereo
411  if (numChannels < 1 ||numChannels > 2)
412  {
413  std::cout << "ERROR: this AIFF file seems to be neither mono nor stereo (perhaps multi-track, or corrupted?)" << std::endl;
414  return false;
415  }
416 
417  // check bit depth is either 8, 16 or 24 bit
418  if (bitDepth != 8 && bitDepth != 16 && bitDepth != 24)
419  {
420  std::cout << "ERROR: this file has a bit depth that is not 8, 16 or 24 bits" << std::endl;
421  return false;
422  }
423 
424  // -----------------------------------------------------------
425  // SSND CHUNK
426  int s = indexOfSoundDataChunk;
427  std::string soundDataChunkID (fileData.begin() + s, fileData.begin() + s + 4);
428  int32_t soundDataChunkSize = fourBytesToInt (fileData, s + 4, Endianness::BigEndian);
429  int32_t offset = fourBytesToInt (fileData, s + 8, Endianness::BigEndian);
430  //int32_t blockSize = fourBytesToInt (fileData, s + 12, Endianness::BigEndian);
431 
432  int numBytesPerSample = bitDepth / 8;
433  int numBytesPerFrame = numBytesPerSample * numChannels;
434  int totalNumAudioSampleBytes = numSamplesPerChannel * numBytesPerFrame;
435  int samplesStartIndex = s + 16 + (int)offset;
436 
437  // sanity check the data
438  if ((soundDataChunkSize - 8) != totalNumAudioSampleBytes || totalNumAudioSampleBytes > (fileData.size() - samplesStartIndex))
439  {
440  std::cout << "ERROR: the metadatafor this file doesn't seem right" << std::endl;
441  return false;
442  }
443 
444  clearAudioBuffer();
445  samples.resize (numChannels);
446 
447  for (int i = 0; i < numSamplesPerChannel; i++)
448  {
449  for (int channel = 0; channel < numChannels; channel++)
450  {
451  int sampleIndex = samplesStartIndex + (numBytesPerFrame * i) + channel * numBytesPerSample;
452 
453  if (bitDepth == 8)
454  {
455  int8_t sampleAsSigned8Bit = (int8_t)fileData[sampleIndex];
456  T sample = (T)sampleAsSigned8Bit / (T)128.;
457  samples[channel].push_back (sample);
458  }
459  else if (bitDepth == 16)
460  {
461  int16_t sampleAsInt = twoBytesToInt (fileData, sampleIndex, Endianness::BigEndian);
462  T sample = sixteenBitIntToSample (sampleAsInt);
463  samples[channel].push_back (sample);
464  }
465  else if (bitDepth == 24)
466  {
467  int32_t sampleAsInt = 0;
468  sampleAsInt = (fileData[sampleIndex] << 16) | (fileData[sampleIndex + 1] << 8) | fileData[sampleIndex + 2];
469 
470  if (sampleAsInt & 0x800000) // if the 24th bit is set, this is a negative number in 24-bit world
471  sampleAsInt = sampleAsInt | ~0xFFFFFF; // so make sure sign is extended to the 32 bit float
472 
473  T sample = (T)sampleAsInt / (T)8388608.;
474  samples[channel].push_back (sample);
475  }
476  else
477  {
478  assert (false);
479  }
480  }
481  }
482 
483  return true;
484 }

541 bool AudioFile<T>::saveToWaveFile (std::string filePath)
542 {
543  std::vector<uint8_t> fileData;
544 
545  int32_t dataChunkSize = getNumSamplesPerChannel() * (getNumChannels() * bitDepth / 8);
546 
547  // -----------------------------------------------------------
548  // HEADER CHUNK
549  addStringToFileData (fileData, "RIFF");
550 
551  // The file size in bytes is the header chunk size (4, not counting RIFF and WAVE) + the format
552  // chunk size (24) + the metadata part of the data chunk plus the actual data chunk size
553  int32_t fileSizeInBytes = 4 + 24 + 8 + dataChunkSize;
554  addInt32ToFileData (fileData, fileSizeInBytes);
555 
556  addStringToFileData (fileData, "WAVE");
557 
558  // -----------------------------------------------------------
559  // FORMAT CHUNK
560  addStringToFileData (fileData, "fmt ");
561  addInt32ToFileData (fileData, 16); // format chunk size (16 for PCM)
562  addInt16ToFileData (fileData, 1); // audio format = 1
563  addInt16ToFileData (fileData, (int16_t)getNumChannels()); // num channels
564  addInt32ToFileData (fileData, (int32_t)sampleRate); // sample rate
565 
566  int32_t numBytesPerSecond = (int32_t) ((getNumChannels() * sampleRate * bitDepth) / 8);
567  addInt32ToFileData (fileData, numBytesPerSecond);
568 
569  int16_t numBytesPerBlock = getNumChannels() * (bitDepth / 8);
570  addInt16ToFileData (fileData, numBytesPerBlock);
571 
572  addInt16ToFileData (fileData, (int16_t)bitDepth);
573 
574  // -----------------------------------------------------------
575  // DATA CHUNK
576  addStringToFileData (fileData, "data");
577  addInt32ToFileData (fileData, dataChunkSize);
578 
579  for (int i = 0; i < getNumSamplesPerChannel(); i++)
580  {
581  for (int channel = 0; channel < getNumChannels(); channel++)
582  {
583  if (bitDepth == 8)
584  {
585  uint8_t byte = sampleToSingleByte (samples[channel][i]);
586  fileData.push_back (byte);
587  }
588  else if (bitDepth == 16)
589  {
590  int16_t sampleAsInt = sampleToSixteenBitInt (samples[channel][i]);
591  addInt16ToFileData (fileData, sampleAsInt);
592  }
593  else if (bitDepth == 24)
594  {
595  int32_t sampleAsIntAgain = (int32_t) (samples[channel][i] * (T)8388608.);
596 
597  uint8_t bytes[3];
598  bytes[2] = (uint8_t) (sampleAsIntAgain >> 16) & 0xFF;
599  bytes[1] = (uint8_t) (sampleAsIntAgain >> 8) & 0xFF;
600  bytes[0] = (uint8_t) sampleAsIntAgain & 0xFF;
601 
602  fileData.push_back (bytes[0]);
603  fileData.push_back (bytes[1]);
604  fileData.push_back (bytes[2]);
605  }
606  else
607  {
608  assert (false && "Trying to write a file with unsupported bit depth");
609  return false;
610  }
611  }
612  }
613 
614  // check that the various sizes we put in the metadata are correct
615  if (fileSizeInBytes != (fileData.size() - 8) || dataChunkSize != (getNumSamplesPerChannel() * getNumChannels() * (bitDepth / 8)))
616  {
617  std::cout << "ERROR: couldn't save file to " << filePath << std::endl;
618  return false;
619  }
620 
621  // try to write the file
622  return writeDataToFile (fileData, filePath);
623 }

628 {

632  int32_t numBytesPerFrame = numBytesPerSample * getNumChannels();
633  int32_t totalNumAudioSampleBytes = getNumSamplesPerChannel() * numBytesPerFrame;
634  int32_t soundDataChunkSize = totalNumAudioSampleBytes + 8;
635 
636  // -----------------------------------------------------------
637  // HEADER CHUNK
638  addStringToFileData (fileData, "FORM");
639 
640  // The file size in bytes is the header chunk size (4, not counting FORM and AIFF) + the COMM
641  // chunk size (26) + the metadata part of the SSND chunk plus the actual data chunk size
642  int32_t fileSizeInBytes = 4 + 26 + 16 + totalNumAudioSampleBytes;
643  addInt32ToFileData (fileData, fileSizeInBytes, Endianness::BigEndian);
644 
645  addStringToFileData (fileData, "AIFF");
646 
647  // -----------------------------------------------------------
648  // COMM CHUNK

650  addInt32ToFileData (fileData, 18, Endianness::BigEndian); // commChunkSize
651  addInt16ToFileData (fileData, getNumChannels(), Endianness::BigEndian); // num channels
652  addInt32ToFileData (fileData, getNumSamplesPerChannel(), Endianness::BigEndian); // num samples per channel
653  addInt16ToFileData (fileData, bitDepth, Endianness::BigEndian); // bit depth
654  addSampleRateToAiffData (fileData, sampleRate);
655 
656  // -----------------------------------------------------------
657  // SSND CHUNK
658  addStringToFileData (fileData, "SSND");
659  addInt32ToFileData (fileData, soundDataChunkSize, Endianness::BigEndian);
660  addInt32ToFileData (fileData, 0, Endianness::BigEndian); // offset
661  addInt32ToFileData (fileData, 0, Endianness::BigEndian); // block size
662 
663  for (int i = 0; i < getNumSamplesPerChannel(); i++)
664  {
665  for (int channel = 0; channel < getNumChannels(); channel++)
666  {
667  if (bitDepth == 8)
668  {
669  uint8_t byte = sampleToSingleByte (samples[channel][i]);
670  fileData.push_back (byte);
671  }
672  else if (bitDepth == 16)
673  {
674  int16_t sampleAsInt = sampleToSixteenBitInt (samples[channel][i]);
675  addInt16ToFileData (fileData, sampleAsInt, Endianness::BigEndian);
676  }
677  else if (bitDepth == 24)
678  {
679  int32_t sampleAsIntAgain = (int32_t) (samples[channel][i] * (T)8388608.);
680 
681  uint8_t bytes[3];
682  bytes[0] = (uint8_t) (sampleAsIntAgain >> 16) & 0xFF;
683  bytes[1] = (uint8_t) (sampleAsIntAgain >> 8) & 0xFF;
684  bytes[2] = (uint8_t) sampleAsIntAgain & 0xFF;
685 
686  fileData.push_back (bytes[0]);
687  fileData.push_back (bytes[1]);
688  fileData.push_back (bytes[2]);
689  }
690  else
691  {
692  assert (false && "Trying to write a file with unsupported bit depth");
693  return false;
694  }
695  }
696  }
697 
698  // check that the various sizes we put in the metadata are correct
699  if (fileSizeInBytes != (fileData.size() - 8) || soundDataChunkSize != getNumSamplesPerChannel() * numBytesPerFrame + 8)
700  {
701  std::cout << "ERROR: couldn't save file to " << filePath << std::endl;
702  return false;
703  }
704 
705  // try to write the file
706  return writeDataToFile (fileData, filePath);
707 }
