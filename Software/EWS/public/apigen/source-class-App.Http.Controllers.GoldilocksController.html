    /**
     * Enrolls new listener in the program. Returns error if listener already exists.
     * POST: '/new-participant'
     *
     * @param  Request
     * @return View
     */
    public function newParticipant(Request $request){
        $researcher = Researcher::where('researcher_id', $request->input('researcher_id'))->first();

        if($request->has(['listener_id', 'listener_pin', 'researcher_id'])){

            // Check if listener exists
            if(Listener::where('listener_id', $request->input('listener_id'))->exists()){
                return view('goldilocks.researcher-new-participant', compact('researcher'))->withErrors(['Listener '.$request->input('listener_id').' already exists']);
            }

            // Create a new listener
            $listener = Listener::create([
                'listener_id' => $request->input('listener_id'),
                'pin' => $request->input('listener_pin'),
                'researcher_id' => $researcher->id
            ]);

            session(['listener_id' => $listener->listener_id]);
            return redirect('researcher-goldilocks');
        }

        //form missing data
        return view('goldilocks.researcher-new-participant', compact('researcher'))->withErrors(['Please enter a valid listener id and pin']);
    }

    /**
     * Login function for the researcher.
     * POST: '/attempt-researcher-login'
     *
     * @param  Request
     * @return View

    public function attemptResearcherLogin(Request $request){
        $listener = Listener::where('listener_id', $request->input('listener_id'))->first();
        $researcher = Researcher::where('researcher_id', $request->input('researcher_id'))->first();

        if($request->has(['listener_id', 'listener_pin', 'researcher_id'])){
            if($listener->pin !== $request->input('listener_pin')){
                //pin does not match
                return redirect('researcher-login/'.$researcher->researcher_id.'/'.$listener->listener_id)->withErrors(['Pin does not match']);
            }
            else if($listener->pin === $request->input('listener_pin')){
                session(['listener_id' => $listener->listener_id]);
                session(['researcher_id' => $researcher->researcher_id]);

                return redirect('researcher-goldilocks');
            }
        }

        //form missing data
        return redirect('researcher-login/'.$researcher->researcher_id.'/'.$listener->listener_id)->withErrors(['Pin does not match']);
    }

    /**
     * Saves the program as a new program after user adjustment.
     * POST: '/user-goldilocks'
     *
     * @param  Request
     */
    public function saveProgram(Request $request){
        $listener = Listener::where('listener_id', session('listener_id'))->first();
        $data = $request->json()->all();
        $program = GoldilocksProgram::create([
            'listener_id' => $listener->id,
            'name' => $data[0],
            'parameters' => json_encode($data[1])
        ]);
        $listener->current_program_id = $program->id;
        $listener->save();
    }

    public function updateProgram(Request $request){

        $program = GoldilocksProgram::where('id', $data[0])->first();

    /**
     * Instructs the MHA to set the currently selected program as the current settings.
     * The function is passed through 'program_id' as an argument and it finds the
     * correct program.
     * POST: '/programs/set'
     *
     * @param Request
     */
    public function setProgram(Request $request){
        $data = $request->json()->all();

        $program = GoldilocksProgram::where('id', $data["program_id"])->first();

        $log = Log::create([
            'request_action' => 1,
            'data' => $program->getAPIString(),
            'status' => 0
        ]);

        $socket = socket_create(AF_INET, SOCK_STREAM, SOL_TCP);
        if(socket_connect($socket, 'localhost', '8005') === true){
            $action_update_params = ["REQUEST_ACTION" => 1, "ID" => $log->id];
            socket_write($socket, json_encode($action_update_params), strlen(json_encode($action_update_params)));
        }

        $whileLog = 0;
        $start = microtime(true);
        while(microtime(true) - $start < 4){
            $whileLog = Log::where("id", $log->id)->first();
            if($whileLog->status != 0){
                break 1;
            }
            //sleep for a tenth of a second
            usleep(100000);
        }
        return (json_encode(["log_id" => $whileLog->id, "status" => $whileLog->status]));

    }
